### 2.2 状态流转确认

**主线状态**: 草稿 → 待评审 → 已就绪 → 已纳版 → 已投产 → 已取消

**已纳版子状态**: 开发中 → SIT测试 → UAT测试 → 封板

**分支流转**:
- 待评审 → 已拒绝（可重新编辑后再次提交）
- 已就绪 → 草稿（需求变更）
- 已纳版（非封板）→ 草稿（需求变更）
- 封板 → 草稿（紧急变更，需审批）
- 已投产 → 已就绪（回滚）

### 3.2 状态流转校验规则

```typescript
// 状态流转图定义
const STATUS_TRANSITIONS: Record<ReqStatus, ReqStatus[]> = {
  DRAFT: ['PENDING_REVIEW', 'CANCELLED'],
  PENDING_REVIEW: ['READY', 'REJECTED', 'CANCELLED'],
  READY: ['ONBOARDED', 'CANCELLED', 'DRAFT'], // 需求变更
  REJECTED: ['DRAFT', 'CANCELLED'],
  ONBOARDED: ['READY', 'RELEASED', 'CANCELLED', 'DRAFT'], // 需求变更/紧急变更
  RELEASED: ['READY'], // 回滚
  CANCELLED: [],
};

// 子状态流转（仅ONBOARDED状态下）
const SUB_STATUS_TRANSITIONS: Record<ReqSubStatus, ReqSubStatus[]> = {
  DEV_IN_PROGRESS: ['SIT_TESTING', 'READY', 'DRAFT'],
  SIT_TESTING: ['UAT_TESTING', 'READY', 'DRAFT'],
  UAT_TESTING: ['FROZEN', 'READY', 'DRAFT'],
  FROZEN: ['RELEASED', 'DRAFT'], // 紧急变更
};

// 状态流转校验函数
function canTransition(
  currentStatus: ReqStatus,
  targetStatus: ReqStatus,
  currentSubStatus?: ReqSubStatus,
  targetSubStatus?: ReqSubStatus
): boolean {
  // 主线状态流转校验
  if (!STATUS_TRANSITIONS[currentStatus]?.includes(targetStatus)) {
    return false;
  }
  
  // 子状态流转校验（仅ONBOARDED状态）
  if (currentStatus === 'ONBOARDED' && targetStatus === 'ONBOARDED') {
    if (!currentSubStatus || !targetSubStatus) return false;
    return SUB_STATUS_TRANSITIONS[currentSubStatus]?.includes(targetSubStatus);
  }
  
  return true;
}

// 操作权限校验
const OPERATION_PERMISSIONS: Record<OperationType, Role[]> = {
```

### 6.1 循环依赖检测

```typescript
// 检测需求依赖是否存在循环
function hasCircularDependency(
  sourceReqId: string,
  targetReqId: string,
  existingDeps: Map<string, string[]>
): boolean {
  // DFS检测从targetReqId出发是否能回到sourceReqId
  const visited = new Set<string>();
  const stack = [targetReqId];
  
  while (stack.length > 0) {
    const current = stack.pop()!;
    if (current === sourceReqId) return true;
    if (visited.has(current)) continue;
    visited.add(current);
    
    const deps = existingDeps.get(current) || [];
    stack.push(...deps);
  }
  
  return false;
}
```

### 6.2 依赖风险检查

```typescript
// 检查依赖风险等级
function checkDependencyRisk(dependencyStatus: ReqStatus): {
  level: 'warning' | 'high' | 'critical' | null;
  message: string;
} {
  switch (dependencyStatus) {
    case 'ONBOARDED':
    case 'RELEASED':
      return { level: null, message: '依赖已满足' };
    case 'CANCELLED':
      return { level: 'critical', message: '依赖已取消' };
    case 'DRAFT':
    case 'PENDING_REVIEW':
    case 'REJECTED':
      return { level: 'high', message: '依赖状态未完成' };
    case 'READY':
      return { level: 'warning', message: '依赖未纳入本车' };
    default:
      return { level: 'warning', message: '依赖状态未知' };
  }
}
```
